country_event = {
	id = debug.2
	title = ""
	desc = ""

	is_triggered_only = yes
	
	option = {
		name = "Уничтожение ВНС"
		
		set_variable = { conservatism_pop = party_popularity@despotism }
		set_variable = { progressivism_pop = party_popularity@despotism }
		set_variable = { liberalism_pop = party_popularity@despotism }
		set_variable = { communist_pop = party_popularity@communist }

		multiply_variable = { conservatism_pop = 100 }
		multiply_variable = { progressivism_pop = 100 }
		multiply_variable = { liberalism_pop = 100 }
		multiply_variable = { communist_pop = 100 }

		multiply_variable = { conservatism_pop = 0.5 }
		round_variable = conservatism_pop

		multiply_variable = { progressivism_pop = 0.333 }
		round_variable = progressivism_pop

		subtract_from_variable = { liberalism_pop = conservatism_pop }
		subtract_from_variable = { liberalism_pop = progressivism_pop }

		set_variable = { sum_pop = conservatism_pop }
		add_to_variable = { sum_pop = progressivism_pop }
		add_to_variable = { sum_pop = liberalism_pop }
		add_to_variable = { sum_pop = communist_pop }

		add_to_variable = { sum_pop = -100 }
		multiply_variable = { sum_pop = -1 }

		add_to_variable = { liberalism_pop = sum_pop }

		log = "cons:[?conservatism_pop] - socdem: [?progressivism_pop] - lib: [?liberalism_pop] - com: [?communist_pop] - sum: [?sum_pop]"

		set_popularities = {
			communist = var:communist_pop
			conservatism = var:conservatism_pop
			progressivism = var:progressivism_pop
			liberalism = var:liberalism_pop
		}

	}	

	option = {
		name = "Верная популярность партий"

				set_variable = { communist_pop = party_popularity@communist }
				set_variable = { fascism_pop = party_popularity@fascism }
				set_variable = { despotism_pop = party_popularity@despotism }

				multiply_variable = { communist_pop = 100 }
				multiply_variable = { fascism_pop = 100 }
				multiply_variable = { despotism_pop = 100 }

				set_variable = { sum_pop = communist_pop }
				add_to_variable = { sum_pop = fascism_pop }
				add_to_variable = { sum_pop = despotism_pop }

				add_to_variable = { sum_pop = -100 }
				multiply_variable = { sum_pop = -1 }
				add_to_variable = { despotism_pop = sum_pop }


				log = "com:[?communist_pop] - fasc: [?fascism_pop] - desp: [?despotism_pop] - sum: [?sum_pop]"

				set_popularities = {
					communist = var:communist_pop
					fascism = var:fascism_pop
					paternalism = var:despotism_pop
				}

	}	
}
